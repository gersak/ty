Code Examples

Basic Usage

<!-- Plain (default) -->
            <ty-button flavor="important">Plain</ty-button>

            <!-- Filled -->
            <ty-button filled flavor="positive">Filled</ty-button>

            <!-- Outlined -->
            <ty-button outlined flavor="negative">Outlined</ty-button>

            <!-- Filled + Outlined -->
            <ty-button filled outlined flavor="exception">Both</ty-button>

            <!-- Accent -->
            <ty-button accent flavor="important">Accent</ty-button>

With Slots

<!-- Using Unicode symbols -->
            <ty-button filled outlined flavor="important">
            <span slot="start">➕</span>
            Add Item
            </ty-button>

            <!-- Using SVG with proper namespace -->
            <ty-button filled outlined flavor="important">
            <span slot="start" class="icon">
            <svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 24 24" fill="currentColor">
            <path d="M19 13h-6v6h-2v-6H5v-2h6V5h2v6h6v2z"/>
            </svg>
            </span>
            Add Item
            </ty-button>

            <!-- End slot -->
            <ty-button outlined flavor="neutral">
            Next
            <span slot="end">▶</span>
            </ty-button>

            <!-- Icon only -->
            <ty-button accent flavor="important">
            <span slot="start">☰</span>
            </ty-button>

JavaScript API

// Get element
            const btn = document.querySelector('ty-button');

            // Set properties
            btn.flavor = 'positive';
            btn.disabled = true;
            btn.label = 'New Label';
            btn.size = 'lg';

            // Set appearance attributes
            btn.filled = true;
            btn.outlined = false;
            btn.accent = false;

            // Or use setAttribute for boolean attributes
            btn.setAttribute('filled', '');
            btn.setAttribute('outlined', '');
            btn.removeAttribute('accent');

            // Listen to events
            btn.addEventListener('click', (e) => {
            console.log('Button clicked!', e.detail);
            });
